                        // check if string is palindrome
                        function palinCheck(uString){
                            //make lower case
                            uString = uString.toLowerCase();

                            // ** Remove special chars/spaces **
                            // Will only return values a through z & 0 through 9
                            let regex = /[^a-z0-9]/gi;
                            uString = uString.replace(regex,"");

                            let revString = [];
                            let returnObj = {};

                            for(let i = uString.length -1; i >= 0; i--){
                                revString += uString[i];
                            }

                            if(revString == uString){
                                returnObj.msg = "Well done! You entered a palindrome!";
                                document.getElementById("alert").classList.add("alert-success");
                            } else {
                                returnObj.msg = "Sorry! You didn't enter a palindrome.";
                                document.getElementById("alert").classList.remove("alert-success");
                                document.getElementById("alert").classList.add("alert-danger");
                            }

                            returnObj.reversed = revString;

                            return returnObj;
                        }

                        //display a message to the user
                        function showMessage(returnObj){
                            document.getElementById("alertHeader").innerHTML = returnObj.msg;
                            document.getElementById("msg").innerHTML = `Your phrase reversed is: ${returnObj.reversed}`;
                            document.getElementById("alert").classList.remove("invisible");
                        }

The app receives a string from the input field and utilises a regular expression to make sure that only certain characters are stored in the string variable. It loops over the user string starting from the end so that the new string is the reverse of the user string. This takes advantage of the fact that strings are arrays in javascript. Then it just compares the user string with the reversed string to see if they are the same. If they are, its a palindrome. The user will receive a message of success if they don't enter a palindrome but will get a different message if they do.
